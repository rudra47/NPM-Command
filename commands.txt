1. npm i yarn -g	= install yarn globally
2. yarn global add create-react-app	= install create-react-app through yarn
3. yarn init	= initialize package.json
4. yarn add jquery	= install jquery module
5. yarn add webpack webpack-cli --save-dev	= install webpack and webpack-cli with development mode locally
6. yarn global add webpack	= install webpack globally
7. npm uninstall webpack -g	= uninstall webpack which is globally
